Задание №1.

SELECT c.login, COUNT(o."inDelivery") FROM "Couriers" AS c INNER JOIN "Orders" as o ON c.id = o."courierId" WHERE o."inDelivery" = true GROUP BY c.login;

Задание №2.

SELECT track, (CASE WHEN finished=true THEN 2 WHEN cancelled=true THEN -1 WHEN "inDelivery"=true THEN 1 WHEN (finished or cancelled or "inDelivery")=false THEN 0 END) AS status FROM "Orders";
